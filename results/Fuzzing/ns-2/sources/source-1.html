145 fun plusMinus(expression: String): Int {
146     if (!expression.matches(Regex("(\\d+ ([+\\-]) )*\\d+"))) {
147         throw IllegalArgumentException("Invalid expression")
148     }
149     val list = expression.split(" ")
150     var sum = list[0].toInt()
151 
152     for (i in 2 until list.size step 2) {
153         when (list[i - 1][0]) {
154             '-' -> {
155                 sum -= list[i].toInt()
156             }
157             '+' -> {
158                 sum += list[i].toInt()
159             }
160         }
161     }
162 
163     return sum
164 }
